# Importing necessary modules from browser import document # Function to greet the user def greet_user(event): name = document["name"].value # Get the name entered by the user greeting = f"Hello, {name}!" # Create the greeting message document["greeting"].text = greeting # Display the greeting # Adding event listener to the button document["greet_button"].bind("click", greet_user)
